203STATIC void MTIMER_vInit(u8_t u8Channel, u8_t u8Mode)
204{
205 MTIMER_vStopTimer(u8Channel);
206
207 switch (u8Channel)
208 {
209 case TIMER_CHANNEL_0:
210 switch (u8Mode)
211 {
212 case TIMER_MODE_NORMAL:
213 TCCR0 = TIMER_MODE_NORMAL;
214 SET_BIT(TIMSK, TIMSK_TOIE2);
215 break;
216 case TIMER_MODE_CTC:
217 TCCR0 = (TIMER_CFG_TIMER0_CTC_PORT_OUTPUT_MODE | TIMER_MODE_CTC);
218 SET_BIT(TIMSK, TIMSK_OCIE0);
219 break;
220 case TIMER_MODE_PWM_FAST:
221 TCCR0 = (TIMER_CFG_TIMER0_FASTPWM_PORT_OUTPUT_MODE | TIMER_MODE_PWM_FAST);
222 break;

224 TCCR0 = (TIMER_CFG_TIMER0_PHASECORRECTPWM_PORT_OUTPUT_MODE | TIMER_MODE_PWM_PHASECORRECT);
225 break;
226 default:
227 break;
228 }
229 break;
230 case TIMER_CHANNEL_1A:
231 case TIMER_CHANNEL_1B:
232 switch (u8Mode)
233 {
234 case TIMER_MODE_NORMAL:
235 TCCR1A = 0;
236 TCCR1B = 0;

238 break;
239 case TIMER_MODE_CTC:
240 TCCR1A = (((u8Channel == TIMER_CHANNEL_1A) ? TIMER_CFG_TIMER1A_CTC_PORT_OUTPUT_MODE : TIMER_CFG_TIMER1B_CTC_PORT_OUTPUT_MODE) | 0);
241 TCCR1B = 0x08;
242 SET_BIT(TIMSK, ((u8Channel == TIMER_CHANNEL_1A) ? TIMSK_OCIE1A : TIMSK_OCIE1B));
243 break;
244 case TIMER_MODE_PWM_FAST:
245 TCCR1A = (((u8Channel == TIMER_CHANNEL_1A) ? TIMER_CFG_TIMER1A_FASTPWM_PORT_OUTPUT_MODE : TIMER_CFG_TIMER1B_FASTPWM_PORT_OUTPUT_MODE) | 0x01);
