159STATIC void MTIMER_vInit(u8_t u8Channel, u8_t u8Mode)
160{
161 MTIMER_vStopTimer(u8Channel);
162
163 switch (u8Channel)
164 {
165 case TIMER_CHANNEL_0:
166 TCCR0 = u8Mode;
167 SET_BIT(TIMSK, TIMSK_OCIE0);
168 break;
169 case TIMER_CHANNEL_1A:
170 case TIMER_CHANNEL_1B:
171 switch (u8Mode)
172 {
173 case TIMER_MODE_NORMAL:
174 TCCR1A = 0;
175 TCCR1B = 0;
176 SET_BIT(TIMSK, TIMSK_TOIE1);
177 break;
178 case TIMER_MODE_CTC:
179 TCCR1A = 0;
180 TCCR1B = 0x08;
181 SET_BIT(TIMSK, ((u8Channel == TIMER_CHANNEL_1A) ? TIMSK_OCIE1A : TIMSK_OCIE1B));
182 break;
183 default:
184 break;
185 }
186 break;
187 case TIMER_CHANNEL_2:
188 TCCR2 = u8Mode;
189 SET_BIT(TIMSK, TIMSK_OCIE2);
190 break;
191 default:
192 break;
193 }

234void MTIMER_vDelaySec(u8_t u8Channel, u32_t u32Delay)
235{
236 u8_t L_u8Prescaler = ((u8Channel == TIMER_CHANNEL_2) ? TIMER_PRESCALER_TIMER2_1024 : TIMER_PRESCALER_TIMER01_1024);
237 MTIMER_vInit(u8Channel, TIMER_MODE_CTC);
238
239 switch (u8Channel)
240 {
241 case TIMER_CHANNEL_0:
242 case TIMER_CHANNEL_2:
243 MTIMER_vSetInitialCTCModeValue(u8Channel, 125);
244
245 for (u8_t u8LoopCounter = 0; u8LoopCounter < 125; u8LoopCounter++)
246 {
247 MTIMER_vStartTimerCTC(u8Channel, L_u8Prescaler, u32Delay);
248 }
249 break;
250 case TIMER_CHANNEL_1A:
251 case TIMER_CHANNEL_1B:
252 MTIMER_vSetInitialCTCModeValue(TIMER_CHANNEL_1A, 15625);
253 MTIMER_vStartTimerCTC(TIMER_CHANNEL_1A, L_u8Prescaler, u32Delay);
254 break;
255 default:
256 break;
257 }
258
259 MTIMER_vStopTimer(u8Channel);
