102STATIC void MTIMER_vInit(u8_t u8Channel, u8_t u8Mode, bool_t bIsSync)
103{
104 MTIMER_vStopTimer(u8Channel);
105
106 switch (u8Channel)
107 {
108 case TIMER_CHANNEL_0:
109 TCCR0 = u8Mode;
110 if (bIsSync)
111 {
112 SET_BIT(TIMSK, 1);
113 }
114 else
115 {
116 CLEAR_BIT(TIMSK, 1);
117 }
118 break;
119 case TIMER_CHANNEL_1A:
120 case TIMER_CHANNEL_1B:
121 switch (u8Mode)

123 case TIMER_MODE_NORMAL:
124 TCCR1A = 0;

126

128 {
129 SET_BIT(TIMSK, 2);
130 }

143 }
